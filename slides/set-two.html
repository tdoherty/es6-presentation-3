let s = new Set();

        s.add(1);    // Set {1}
        s.add(2);    // Set {1, 2}
        s.add(3);    // Set {1, 2, 3}
        s.size;      // 3
        s.has(3);    // true
        [...s]       // [1, 2, 3]
        s.add(3);    // Set {1, 2, 3}
        s.delete(3); // Set {1, 2}
        s.has(3);    // false
        s.clear();   // Set {}
        s.size;      // 0
