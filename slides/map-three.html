    let strKey = 'stringy';

    m.get(objKey);    // "value for objKey"
    m.get(fnKey);     // "value for fnKey"

    m.get('stringy'); // "value for 'stringy'"
    m.get({});        // undefined
    m.get(() => {});  // undefined
