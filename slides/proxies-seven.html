let tracer = {
  get(target, trapName, receiver) {
    return function (...args) {
      console.log(trapName.toUppercase() +
                  ' ' +
                  args.slice(1));
      return Reflect[trapName](...args);
    }
  }
};

let p = new Proxy({}, tracer);

p.foo = 'bar' // SET foo,'bar',[object Object]
p.foo         // GET foo,[object Object]
